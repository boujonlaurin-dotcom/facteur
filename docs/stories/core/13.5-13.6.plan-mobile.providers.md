  factory LearningProposal.fromJson(Map<String, dynamic> json) { ... }

  /// Phrase factuelle pour la ligne (règle spec format-visuel §Règles de phrase).
  String justificationPhrase() {
    switch (proposalType) {
      case ProposalType.sourcePriority:
        return (proposedValue ?? 0) < (currentValue ?? 0)
            ? 'rarement ouvert'
            : 'souvent ouvert';
      case ProposalType.muteEntity:   return 'vue mais ignorée';
      case ProposalType.followEntity: return 'suivie de près';
      case ProposalType.unknown:      return '';
    }
  }
}

## 2. Repository

Fichier : `apps/mobile/lib/features/learning_checkpoint/repositories/learning_checkpoint_repository.dart`

```dart
class LearningCheckpointRepository {
  final ApiClient _api;
  LearningCheckpointRepository(this._api);

  /// GET /learning-proposals — retourne [] sur 404/timeout (silent fail).
  /// Timeout spec : 300ms p95 côté backend ; côté client on laisse Dio par défaut
  /// mais on peut configurer un Options(sendTimeout: ..., receiveTimeout: 2s).
  Future<List<LearningProposal>> fetchProposals() async {
    try {
      final res = await _api.dio.get<dynamic>(
        'learning-proposals',
        options: Options(
          sendTimeout: const Duration(seconds: 2),
          receiveTimeout: const Duration(seconds: 2),
        ),
      );
      final data = res.data as List? ?? [];
      return data
          .whereType<Map<String, dynamic>>()
          .map(LearningProposal.fromJson)
          .toList();
    } on DioException catch (e) {
      // 404 / 500 / timeout → carte non affichée, pas d'erreur UI
      _logNonFatal(e);
      return [];
    }
  }

  /// POST /apply-proposals
  Future<ApplyProposalsResponse> applyProposals(List<ApplyAction> actions) async {
    final res = await _api.dio.post<Map<String, dynamic>>(
      'apply-proposals',
      data: {'actions': actions.map((a) => a.toJson()).toList()},
    );
    return ApplyProposalsResponse.fromJson(res.data ?? {});
  }
}

final learningCheckpointRepositoryProvider =
    Provider<LearningCheckpointRepository>((ref) {
  return LearningCheckpointRepository(ref.read(apiClientProvider));
});
```

**Remarques** :
- Pas de retry explicite : l'interceptor Dio global gère déjà 401/403 refresh. Pour 5xx on laisse échouer silencieusement (carte hidden).
- `_logNonFatal` : logger existant (à câbler sur `AnalyticsService.trackError` ou `print` en dev — TBD avec convention projet).

## 3. Providers Riverpod

Fichier : `apps/mobile/lib/features/learning_checkpoint/providers/learning_checkpoint_provider.dart`

### 3.1 Provider principal — `learningCheckpointProvider`

Type : `AsyncNotifierProvider<LearningCheckpointNotifier, LearningCheckpointState>`

**Responsabilités** :
1. Appel `repository.fetchProposals()` au build
2. Application du **gating côté client** :
   - feature flag `LearningCheckpointFlags.enabled`
   - override dev : `learning_checkpoint_force_disabled` SharedPreferences
   - cooldown 24h (lire `learningCheckpointCooldownProvider`)
   - flag 1/session (lire `learningCheckpointShownThisSessionProvider`)
   - `proposals.length >= 3`
   - `proposals.map((p) => p.signalStrength).reduce(max) >= 0.6`
3. Tri `signal_strength DESC`, limite 5
4. Exposition d'un `LearningCheckpointState` consommable par le widget
5. Méthodes `dismissItem`, `validate`, `snooze`

### 3.3 `LearningCheckpointNotifier`

```dart
class LearningCheckpointNotifier
    extends AsyncNotifier<LearningCheckpointState> {

  @override
  Future<LearningCheckpointState> build() async {
    final flags = LearningCheckpointFlags;

    // Kill-switches
    if (!flags.enabled) return const LcHidden();
    if (await _isForceDisabled()) return const LcHidden();

    // Cooldown 24h
    final cooldownActive = await ref.watch(
      learningCheckpointCooldownProvider.future,
    );
    if (cooldownActive) return const LcHidden();

    // 1/session
    final shownThisSession = ref.watch(
      learningCheckpointShownThisSessionProvider,
    );
    if (shownThisSession) return const LcHidden();

    // Fetch
    final repo = ref.read(learningCheckpointRepositoryProvider);
    final all = await repo.fetchProposals();

    // Gating pertinence
    if (all.length < flags.minProposals) return const LcHidden();
    final maxSignal = all.map((p) => p.signalStrength).fold<double>(
      0, (a, b) => a > b ? a : b,
    );
    if (maxSignal < flags.minSignalStrength) return const LcHidden();

    // Tri + troncature
    final sorted = [...all]..sort(
      (a, b) => b.signalStrength.compareTo(a.signalStrength),
    );
    final displayed = sorted.take(flags.maxProposalsDisplayed).toList();

    // Mark « shown this session » (side-effect contrôlé)
    ref.read(learningCheckpointShownThisSessionProvider.notifier).state = true;

    return LcVisible(
      displayed: displayed,
      dismissedIds: const {},
      modifiedValues: const {},
      expandedRowId: null,
    );
  }

  // --- Actions ---

  void toggleExpanded(String proposalId) { ... }

  void dismissItem(String proposalId) {
    // Ajoute dans dismissedIds. Si < 1 proposition restante → snooze auto (spec).
  }

  void modifyValue(String proposalId, num newValue) { ... }

  Future<void> validate() async {
    // 1. Passe state à LcApplying
    // 2. Construit List<ApplyAction> :
    //    - proposals dans dismissedIds → action dismiss
    //    - proposals dans modifiedValues → action modify(value)
    //    - autres (pré-actives non touchées) → action accept
    // 3. Appel repository.applyProposals
    // 4. Pose cooldown timestamp
    // 5. Analytics validate
    // 6. state = LcApplied (puis LcHidden après snackbar)
  }

  Future<void> snooze() async {
    // 1. state = LcApplying
    // 2. Toutes les propositions restantes (non dismissed) → action dismiss
    // 3. POST apply-proposals
    // 4. Pose cooldown timestamp
    // 5. Analytics snooze
    // 6. state = LcSnoozed (puis LcHidden)
  }

  Future<void> _markCooldown() async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.setInt(
      'learning_checkpoint_last_action_at',
      DateTime.now().millisecondsSinceEpoch,
    );
    ref.invalidate(learningCheckpointCooldownProvider);
  }
}

final learningCheckpointProvider = AsyncNotifierProvider<
    LearningCheckpointNotifier, LearningCheckpointState>(
  LearningCheckpointNotifier.new,
);
```

### 3.4 `learningCheckpointCooldownProvider`

Fichier : `providers/learning_checkpoint_cooldown_provider.dart`

```dart
final learningCheckpointCooldownProvider = FutureProvider<bool>((ref) async {
  final prefs = await SharedPreferences.getInstance();
  final lastActionAt =
      prefs.getInt('learning_checkpoint_last_action_at');
  if (lastActionAt == null) return false;
  final elapsed = DateTime.now().millisecondsSinceEpoch - lastActionAt;
  return elapsed < LearningCheckpointFlags.cooldown.inMilliseconds;
});
```

## 5. Invalidations cross-providers

Après `validate` ou `snooze` :
- `ref.invalidate(learningCheckpointCooldownProvider)` — pour recalculer le cooldown
- `ref.invalidate(feedProvider)` — pour refresh le feed avec les nouvelles préférences utilisateur (cf. spec § Valider : « feed se rafraîchit »)
- `ref.invalidate(userSourcesProvider)` — si des `source_priority` ont été modifiées
- `ref.invalidate(customTopicsProvider)` — si des `follow_entity` / `mute_entity` ont été appliqués

Liste à affiner en phase CODE selon le comportement réel de l'API (réponse `updated_preferences`).
